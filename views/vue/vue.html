# 使用webpack创建Vue项目

# 初始化



 

  # 创建一个目录，并进行初始化
  cd demo
  npm init -y

# 安装webpack


 








  # 安装webpack webpack-cli webpack-dev-server html-webpack-plugin

# 配置webpack

  # 1、创建配置文件 webpack.config.js 

  # 2、配置webpack.config.js
  const path = require('path')
  const HtmlWebpackPlugin = require('html-webpack-plugin')
  module.exports = {
    entry: './src/main.js', // 入口文件 需要创建src目录并创建main.js文件
    output: {                  // 配置出口文件，这里需要用到node的path
      filename: 'bundle.js', // 出口文件
      path: path.resolve(__dirname, 'dist')  // 出口文件路径
    },
    plugins: [
      new HtmlWebpackPlugin({
        template: path.resolve(__dirname, 'public/index.html') // 指定index.html位置
      })
    ]
  }
  # 3、public/index.html文件
  <!DOCTYPE html>
  <html lang="en">
  <head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>App</title>
  </head>
  <body>
      <div id="app"></div>
  </body>
  </html>
  
  # 4、最后目录如下
  demo
  ├─ package-lock.json
  ├─ package.json
  ├─ public
  │  └─ index.html
  ├─ src
  │  └─ main.js
  └─ webpack.config.js

# 安装Vue 配置main.js


 


























  # 安装Vue 
  npm install vue -S

  # 配置main.js
  import Vue from 'vue'
  import App from './App.vue'

  new Vue({
    render: h => h(App)
  }).$mount('#app')

  # 创建App.vue, 在src目录下创建App.vue
  # App.vue
  <template>
    <div>{{msg}}</div>
  </template>

  <script>
  export default {
    name: 'app',
    data(){
      return {
        msg: '这是一个测试项目'
      }
    }
  }
  </script>

# 安装vue-loader css-loader vue-style-loader vue-template-compiler

官方说明



 










 










 
 






 



  # webpack无法打包.vue文件，需要使用vue-loader来进行打包，
  # vue-loader使用需要手动安装一个css-loader
  npm install vue-loader css-loader vue-style-loader vue-template-compiler --save-dev
  "devDependencies": {
    "css-loader": "^4.2.1",
    "vue-loader": "^15.9.3",
    "vue-style-loader": "^4.1.2",
    "vue-template-compiler": "^2.6.12",
  },

  # 配置webpack.config.js
  const path = require('path')
  const HtmlWebpackPlugin = require('html-webpack-plugin')
  const VueLoaderPlugin = require('vue-loader/lib/plugin')

  module.exports = {
    mode: 'production',
    entry: './src/main.js', // 入口文件 需要创建src目录并创建main.js文件
    output: { // 配置出口文件，这里需要用到node的path
      filename: 'bundle.js', // 出口文件
      path: path.resolve(__dirname, 'dist')  // 出口文件路径
    },
    module: {
      rules: [
        { test: /\.css$/, use: ['vue-style-loader', 'css-loader'] },//处理css文件的规则
        { test: /\.vue$/, use: 'vue-loader' },
      ]
    },
    plugins: [
      new HtmlWebpackPlugin({
        template: path.resolve(__dirname, 'public/index.html') // 指定index.html位置
      }),
      new VueLoaderPlugin()
    ]
  }
